# VS 코드 전용 - 데이터 자동 주입 및 index.html 생성 스크립트 (make_html.py) import os import urllib.request # 1. 파일 존재 여부 확인 (동일 폴더 기준) file_name = 'data.json' if not os.path.exists(file_name): raise FileNotFoundError(f"현재 폴더 내에 {file_name} 파일이 존재하지 않습니다. data.json 파일이 스크립트와 같은 폴더에 있는지 확인하십시오.") with open(file_name, 'r', encoding='utf-8') as f: json_data = f.read() # 2. 깃허브 원본 템플릿 실시간 다운로드 raw_html_url = "https://raw.githubusercontent.com/hwangjuyeong40-wq/kbeauty-lab-dashboard/main/index.html" try: response = urllib.request.urlopen(raw_html_url) base_html = response.read().decode('utf-8') except Exception as e: print(f"원격 템플릿 다운로드 실패로 기본 구조 적용: {e}") base_html = """K-Beauty Insight Lab""" # 3. 데이터 치환 주입 작업 if "const globalDataset = " in base_html: parts = base_html.split("const globalDataset = ") sub_parts = parts[1].split(";", 1) final_html = parts[0] + "const globalDataset = " + json_data + ";" + sub_parts[1] else: final_html = base_html.replace("__DATA__", json_data) # 4. 결과 파일을 index.html 이라는 이름으로 로컬에 바로 저장 (파일명 불일치 문제 해결) output_name = 'index.html' with open(output_name, 'w', encoding='utf-8') as f: f.write(final_html) print(f"작업 완료! 동일 폴더 내에 데이터가 주입된 '{output_name}' 파일이 성공적으로 덮어씌워졌거나 생성되었습니다.")
